CREATE TABLE ITEM
(
ITEMCODE VARCHAR(10) PRIMARY KEY
, ITEMNAME VARCHAR(200) NOT NULL
, MANUFACTURERNAME VARCHAR(200)
, UNITPRICE numeric(10,2) CHECK(UNITPRICE>0)
, MANUFACTURINGYEAR DATE
, ITEMCATEGORY VARCHAR(20)
);

CREATE TABLE CUSTOMER
(
CUSTOMERID numeric PRIMARY KEY
, CUSTOMERNAME VARCHAR(200) NOT NULL
, ADDRESS VARCHAR(300)
, PHONENUMBER VARCHAR(10)
, CUSTOMERMAILID VARCHAR(20)
);

CREATE TABLE ORDERMASTER
(
ORDERID numeric PRIMARY KEY
, ORDERDATE DATE NOT NULL
, TOTALORDERAMOUNT NUMERIC(10,2) CHECK(TOTALORDERAMOUNT>0)
, CUSTOMERID NUMERIC REFERENCES CUSTOMER(CUSTOMERID)
);

ALTER TABLE ORDERMASTER CHANGE TOTALORDERAMOUNT TOTALORDERAMOUNT NUMERIC(10,2) NOT NULL);

ALTER TABLE ORDERMASTER CHANGE CUSTOMERID CUSTOMERID NUMERIC NOT NULL);

CREATE TABLE ORDERTRANSACTION
(
ORDERID NUMERIC 
, ITEMCODE VARCHAR(10)
, QTYORDERED NUMERIC(2) CHECK(QTYORDERED>0)
, PRIMARY KEY (ORDERID, ITEMCODE),
FOREIGN KEY (ITEMCODE) REFERENCES ITEM(ITEMCODE),
FOREIGN KEY (ORDERID) REFERENCES ORDERMASTER(ORDERID)
);
